sudo iptables -A INPUT -p tcp --dport 8080 -j ACCEPT - дозволити доступ до порту 8080(HTTP)

iptables -L -n -v - перевірки активних правил

iptables -A INPUT -p tcp --dport 22 -s 192.168.1.100 -j ACCEPT - відкривати порт конкретній IP-адресі

sudo iptables -L INPUT -n --line-numbers - перегляд правил за номерами

sudo iptables -D INPUT [rule number] - видалення правила 

/sudo/ iptables -A INPUT -p tcp --dport 22 -s 192.168.1.199 -j DROP - заборона IP-адреси на порт
